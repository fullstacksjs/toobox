# isString

Check whether the given value is string or not

### Import

```typescript copy
import { isString } from '@fullstacksjs/toolbox';
```

### Signature

```typescript copy
const isString = (x: unknown): x is string => {}
```

### Examples

```typescript copy
isString('')        // true
isString(undefined) // false
isString(null)      // false
isString(1)         // false
isString(['1'])     // false
```
