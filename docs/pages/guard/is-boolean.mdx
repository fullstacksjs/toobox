# isBoolean

Checks whether the given value is boolean or not

### Import

```typescript copy
import { isBoolean } from '@fullstacksjs/toolbox';
```

### Signature

```typescript copy
function isBoolean(x: unknown): x is boolean {}
```

### Examples

```typescript copy
isBoolean(true); // true
isBoolean(false); // true
isBoolean(''); // false
isBoolean(undefined); // false
isBoolean(null); // false
isBoolean(1); // false
isBoolean(['1']); // false
```
