# isIterable

Check whether the given value is iteratable or not

### Import

```typescript copy
import { isIterable } from '@fullstacksjs/toolbox';
```

### Signature

```typescript copy
const isIterable = <T>(x: unknown): x is Iterable<T> => {}
```

### Examples

```typescript copy
isIterable([])        // true
isIterable(new Map()) // true
isIterable('some')    // true
isIterable({})        // false
isIterable(undefined) // false
```
