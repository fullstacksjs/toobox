# isTruthy

Check whether the given value is truthy

### Import

```typescript copy
import { isTruthy } from '@fullstacksjs/toolbox';
```

### Signature

```typescript copy
function isTruthy<T>(x: T): x is Truthy<T> {};
```

### Examples

```typescript copy
isTruthy(undefined) // false
isTruthy(NaN)    // false
isTruthy(false)  // false
isTruthy(0)      // false
isTruthy(null)   // false
isTruthy('null') // true
isTruthy(1)      // true
isTruthy([])     // true
isTruthy({})     // true
```
