# isNullOrEmpty

Checks whether the given value is null, undefined or empty string.

### Import

```typescript copy
import { isNullOrEmpty } from '@fullstacksjs/toolbox';
```

### Signature

```typescript copy
const isNullOrEmpty = (x: unknown): x is string => {}
```

### Examples

```typescript copy
isNullOrEmpty(null)      // true
isNullOrEmpty(undefined) // true
isNullOrEmpty('')        // true
isNullOrEmpty('f')       // false
```
