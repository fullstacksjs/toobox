# removeTrailingSlashes

Removes trailing slashes from a string

### Import

```typescript copy
import { removeTrailingSlashes } from '@fullstacksjs/toolbox';
```

### Signature

```typescript copy
function removeTrailingSlashes(str: string): string {}
```

### Examples

```typescript copy
removeTrailingSlashes('')           // ''
removeTrailingSlashes('/')          // ''
removeTrailingSlashes('string')     // 'string'
removeTrailingSlashes('/string')    // 'string'
removeTrailingSlashes('//string')   // 'string'
removeTrailingSlashes('//string/')  // 'string/'
removeTrailingSlashes('//string/a') // 'string/a'
```
